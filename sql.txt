CREATE MATERIALIZED VIEW IF NOT EXISTS user_report_by_time AS
SELECT report_by_id, report_at, report_id, middle_type, small_type,
       report_location, report_content, report_lat, report_lot,
       visible, delete_vote, vote_id
FROM user_report
WHERE report_by_id IS NOT NULL AND report_at IS NOT NULL AND report_id IS NOT NULL
PRIMARY KEY ((report_at), report_id, report_by_id);
