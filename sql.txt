CREATE TABLE rtd_db_old AS SELECT * FROM rtd_db;

CREATE TABLE rtd_db_old (
    rtd_code int,
    rtd_time timestamp,
    id uuid,
    latitude double,
    longitude double,
    regioncode bigint,
    rtd_loc text,
    rtd_details list<text>,
    PRIMARY KEY ((rtd_code, rtd_time), id)
);

CREATE TABLE rtd_db_new (
    id uuid,
    rtd_time timestamp,
    rtd_loc text,
    rtd_details list<text>,
    rtd_code int,
    regioncode bigint,
    latitude double,
    longitude double,
    PRIMARY KEY ((rtd_time), id)
) WITH CLUSTERING ORDER BY (id DESC);


DROP TABLE rtd_db;

CREATE TABLE rtd_db (
    id uuid,
    rtd_time timestamp,
    rtd_loc text,
    rtd_details list<text>,
    rtd_code int,
    regioncode bigint,
    latitude double,
    longitude double,
    PRIMARY KEY ((rtd_time), id)
) WITH CLUSTERING ORDER BY (id DESC);
